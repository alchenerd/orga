** Collect

Collects raw inputs from the Org buffer and shape them into
`(:role :content)` lists: chat history, system prompt.
Also collects tool schemas.

#+NAME: orga_core__collect
#+BEGIN_SRC emacs-lisp
(defun orga/subtree-body ()
  "Return the message body of the subtree whose heading is at point.
Skips the :PROPERTIES: drawer and strips any :REASONING:...:END: drawers
so internal chain-of-thought never leaks back into the API context.
Result is trimmed."
  (save-excursion
    (forward-line 1)
    ;; Skip :PROPERTIES:...:END:
    (when (looking-at "[ \t]*:PROPERTIES:")
      (re-search-forward "^[ \t]*:END:" nil t)
      (forward-line 1))
    ;; Collect the raw subtree body
    (let* ((beg (point))
           (end (progn (org-end-of-subtree t) (point)))
           (raw (buffer-substring-no-properties beg end)))
      ;; Strip :REASONING:...:END: blocks (non-greedy, multiline)
      (string-trim
       (replace-regexp-in-string
        ":REASONING:[\0-\377[:nonascii:]]*?:END:\n?" "" raw)))))

(defun orga/collect-history ()
  "Return the full chat history as a vector of OpenAI-style message alists.
Scans ** User, ** Input, ** Assistant, ** Assistant Tool Call, and ** Tool
headings inside * Chat History.
All Org metadata (property drawers, :REPLY: wrappers) is stripped so only
plain content reaches the API.

For ** Assistant Tool Call headings the :TOOL_CALLS_JSON: property is
decoded back into the tool_calls array expected by the API.

For ** Tool headings the :TOOL_CALL_ID: and :FUNCTION_NAME: properties
supply the metadata; the subtree body is the tool result content."
  (let (messages)
    (save-excursion
      (goto-char (point-min))
      (unless (re-search-forward "^\\* Chat History" nil t)
        (error "OrgA: no '* Chat History' section found"))
      (let ((section-end (save-excursion (org-end-of-subtree t) (point))))
        ;; Honor the latest :cutoff: marker: turns at or above it are
        ;; represented by * Summary, so begin collecting after its subtree.
        (let ((cutoff-pos
               (save-excursion
                 (let (pos)
                   (while (re-search-forward "^\\*\\* .*:cutoff:" section-end t)
                     (setq pos (match-beginning 0)))
                   pos))))
          (when cutoff-pos
            (goto-char cutoff-pos)
            (org-end-of-subtree t)))
        (while (re-search-forward
                "^\\*\\* \\(User\\|Input\\|Assistant Tool Call\\|Assistant\\|Tool\\)"
                section-end t)
          (let* ((kind    (match-string 1))
                 (props   (save-excursion
                            (forward-line 1)
                            (when (looking-at "[ \t]*:PROPERTIES:")
                              (let ((drawer-start (point))
                                    (drawer-end   (progn
                                                    (re-search-forward "^[ \t]*:END:" nil t)
                                                    (point))))
                                (let ((text (buffer-substring-no-properties
                                             drawer-start drawer-end)))
                                  text)))))
                 (get-prop (lambda (name)
                             (when props
                               (when (string-match
                                      (concat ":" name ":[ \t]+\\([^\n]+\\)") props)
                                 (string-trim (match-string 1 props))))))
                 (content (orga/subtree-body)))
            (pcase kind
              ;; Plain user message
              ((or "User" "Input")
               (unless (string-empty-p content)
                 (push `((role . "user") (content . ,content)) messages)))

              ;; Regular assistant text reply
              ("Assistant"
               (unless (string-empty-p content)
                 (push `((role . "assistant") (content . ,content)) messages)))

              ;; Assistant message that contained tool calls -- reconstruct
              ;; the tool_calls array from the stored JSON property.
              ("Assistant Tool Call"
               (let* ((tc-json (funcall get-prop "TOOL_CALLS_JSON"))
                      (json-object-type 'alist)
                      (json-array-type  'vector)
                      (tc-parsed (when tc-json
                                   (ignore-errors (json-read-from-string tc-json)))))
                 (when tc-parsed
                   (push `((role       . "assistant")
                           (content    . :null)
                           (tool_calls . ,tc-parsed))
                         messages))))

              ;; Tool result message
              ("Tool"
               (let ((call-id (funcall get-prop "TOOL_CALL_ID"))
                     (fn-name (funcall get-prop "FUNCTION_NAME")))
                 (when (and call-id (not (string-empty-p content)))
                   (push `((role         . "tool")
                           (tool_call_id . ,call-id)
                           (name         . ,(or fn-name ""))
                           (content      . ,content))
                         messages)))))))))
    (vconcat (nreverse messages))))

(defun orga/collect-system-prompt ()
  "Return the body of the #+BEGIN_SRC markdown block inside * System Prompt."
  (save-excursion
    (goto-char (point-min))
    (when (re-search-forward "^\\* System Prompt" nil t)
      (let ((section-end (save-excursion (org-end-of-subtree t) (point))))
        (when (re-search-forward "#\\+BEGIN_SRC markdown" section-end t)
          (forward-line 1)
          (let* ((beg (point))
                 (end (progn
                        (re-search-forward "#\\+END_SRC" section-end t)
                        (match-beginning 0)))
                 (text (string-trim (buffer-substring-no-properties beg end))))
            (unless (string-empty-p text)
              (vector `((role . "developer")
                        (content . ,text))))))))))

(defun orga/collect-summary ()
  "Return the body of the * Summary heading as a single user-message vector.
Returns nil when the heading is absent or empty.  The summary is a
cumulative, disposable recap of the turns above the :cutoff: marker; it is
sent as a `user' message (not `developer') so it stays subordinate to the
live system prompt and the fresh conversation turns."
  (save-excursion
    (goto-char (point-min))
    (when (re-search-forward "^\\* Summary" nil t)
      (let* ((beg (progn (forward-line 1) (point)))
             (end (progn (org-end-of-subtree t) (point)))
             (text (string-trim (buffer-substring-no-properties beg end))))
        (unless (string-empty-p text)
          (vector `((role . "user")
                    (content . ,(concat
                                 "Summary of earlier conversation, condensed "
                                 "to save context. Treat as established "
                                 "background:\n\n" text)))))))))

(defun orga/collect-tools ()
  "Parse the * Tools section and return a vector of tool schema alists.
Each ** heading whose name matches a `orga_tool_schema__*' src block is
included, unless the heading carries a :disabled: tag, in which case the
entire subtree is skipped.
The src block body is evaluated as Emacs Lisp to produce the alist.
Returned as a vector (not a list) so json-serialize unambiguously treats
it as a JSON array -- a list of alists would otherwise be misread as a
single alist, since the first element (itself a list of conses) looks
like a valid (key . value) pair to the alist/array auto-detector."
  (let (tools)
    (save-excursion
      (goto-char (point-min))
      (unless (re-search-forward "^\\* Tools" nil t)
        (error "OrgA: no '* Tools' section found"))
      (let ((section-end (save-excursion (org-end-of-subtree t) (point))))
        (while (re-search-forward "^\\*\\* " section-end t)
          (let* ((heading-line (thing-at-point 'line t))
                 (disabled (string-match ":disabled:" heading-line))
                 (subtree-end (save-excursion (org-end-of-subtree t) (point))))
            (unless disabled
              ;; Look for an orga_tool_schema__* named src block in this subtree
              (save-excursion
                (while (re-search-forward
                        "#\\+NAME:[ \t]+\\(orga_tool_schema__[^\n]+\\)" subtree-end t)
                  (let ((block-name (string-trim (match-string 1))))
                    ;; Find the #+BEGIN_SRC immediately following the #+NAME:
                    (when (re-search-forward "#\\+BEGIN_SRC emacs-lisp" subtree-end t)
                      (forward-line 1)
                      (let* ((beg (point))
                             (end (progn
                                    (re-search-forward "#\\+END_SRC" subtree-end t)
                                    (match-beginning 0)))
                             (src (buffer-substring-no-properties beg end))
                             (schema (ignore-errors (eval (read src)))))
                        (when schema
                          (push schema tools))))))))
            ;; Advance past this subtree
            (goto-char subtree-end)))))
    (let ((collected (nreverse tools)))
      (when collected (vconcat collected)))))

(defun orga/read-input-heading ()
  "Find the last ** Input heading and return (start end text), or nil.
START is the position of the heading line; END is the end of the subtree.
TEXT is the trimmed body (excluding any property drawer)."
  (save-excursion
    (goto-char (point-max))
    (when (re-search-backward "^\\*\\* Input" nil t)
      (let* ((start (point))
             (text  (orga/subtree-body))
             (end   (progn (org-end-of-subtree t) (point))))
        (list start end text)))))

** Assemble

Takes slots and concatenates them into the message vector sent to the API.

#+NAME: orga_core__assemble
#+BEGIN_SRC emacs-lisp
(defvar orga/context-slot-order '(:system :summary :history)
  "Ordered list of context slot keywords.
Slots are concatenated left-to-right to form the final message vector
sent to the API.  Slots whose value is nil or an empty vector are skipped.")

(defun orga/create-context ()
  "Build and return the full context window as a vector of message alists.

Fills a plist of named slots, then concatenates them in the order
declared by `orga/context-slot-order'.

Current slots:
  :system   – persistent system prompt (empty for now)
  :history  – chat history collected from * Chat History"
  (let (slots) ; plist: (:slot-name vector-of-messages ...)

    ;; ---- populate slots -------------------------------------------------

    ;; :system -- persistent instructions sent before everything else.
    (when-let ((sys (orga/collect-system-prompt)))
      (setq slots (plist-put slots :system sys)))

    ;; :summary -- cumulative, disposable recap of pre-cutoff turns.
    ;; Sent as a user message so it never outranks fresh turns.
    (when-let ((sum (orga/collect-summary)))
      (setq slots (plist-put slots :summary sum)))

    ;; :history -- the conversation so far, always present.
    (setq slots (plist-put slots :history (orga/collect-history)))

    ;; ---- assemble in declared order -------------------------------------
    (apply #'vconcat
           (delq nil
                 (mapcar (lambda (key)
                           (let ((val (plist-get slots key)))
                             (and val (not (zerop (length val))) val)))
                         orga/context-slot-order)))))

(defun orga/read-env-file (var-name)
  "Return VAR-NAME's value from a .env file in `default-directory', or nil."
  (let ((env-file (expand-file-name ".env" default-directory)))
    (when (file-exists-p env-file)
      (with-temp-buffer
        (insert-file-contents env-file)
        (goto-char (point-min))
        (when (re-search-forward
               (concat "^" (regexp-quote var-name) "=\\(.*\\)$") nil t)
          (string-trim (match-string 1)))))))

(defun orga/active-key ()
  "Return the OpenRouter API key, trying three sources in order:
1. The openrouter_key Org block (if non-empty).
2. OPENROUTER_API_KEY in a .env file next to this file.
3. OPENROUTER_API_KEY in the process environment."
  (let ((block-key <<openrouter_key>>))
    (cond
     ((and (stringp block-key) (not (string-empty-p block-key))) block-key)
     ((orga/read-env-file "OPENROUTER_API_KEY"))
     ((getenv "OPENROUTER_API_KEY"))
     (t (error "OrgA: no API key found (Org block, .env, or environment)")))))

(cl-defun orga/send-request (model messages &optional tools dry-run)
  "POST MESSAGES to the OpenRouter chat completions endpoint for MODEL.
TOOLS is an optional list of tool-schema alists to include in the payload.
When DRY-RUN is non-nil, pretty-print the payload to *OrgA Dry Run* instead
of sending it and return nil.

Returns a parsed alist. Signals an error on HTTP or parse failure."
  (let* ((payload-alist `((model     . ,model)
                          (reasoning . ((enabled . t)))
                          (messages  . ,messages)
                          ,@(when tools `((tools . ,tools)))))
         (payload (json-serialize payload-alist :false-object :json-false)))
    (if dry-run
        (progn
          (with-current-buffer (get-buffer-create "*OrgA Dry Run*")
            (erase-buffer)
            (insert payload)
            (json-pretty-print-buffer)
            (goto-char (point-min))
            (display-buffer (current-buffer)))
          nil)
      ;; ---------------------------------------------------------------------
      ;; Pure Native Curl Request (WSL / Linux Safe)
      ;; ---------------------------------------------------------------------
      (let ((response-buffer (generate-new-buffer " *orga-curl-res*"))
            (curl-path (executable-find "curl"))
            exit-code)
        (unless curl-path
          (error "OrgA Error: 'curl' executable not found in your environment path"))

        (unwind-protect
            (let ((coding-system-for-write 'utf-8)
                  (coding-system-for-read 'utf-8))

              ;; Fire synchronous curl execution directly into our temporary buffer
              (setq exit-code
                    (apply #'call-process curl-path nil response-buffer nil
                           (list "-s" "-i" "-X" "POST" "https://openrouter.ai/api/v1/chat/completions"
                                 "-H" "Content-Type: application/json; charset=utf-8"
                                 "-H" (concat "Authorization: Bearer " (orga/active-key))
                                 "-d" payload)))

              (with-current-buffer response-buffer
                (goto-char (point-min))
                ;; Validate HTTP Status Code
                (unless (re-search-forward "^HTTP/[0-9.]+\\s-+2[0-9][0-9]" nil t)
                  (let ((error-message (buffer-substring (point-min) (line-end-position))))
                    (error "OrgA: API server error or connection failure -- %s" error-message)))

                ;; Advance past HTTP Header headers safely
                (goto-char (point-min))
                (if (re-search-forward "^\r?$" nil t)
                    (forward-line 1)
                  (error "OrgA: Found no end of HTTP headers in response"))

                ;; Read and return the raw JSON string payload natively
                (let ((json-object-type 'alist)
                      (json-array-type  'vector))
                  (json-read))))
          ;; Clean up our temporary buffer instance cleanly
          (when (buffer-live-p response-buffer)
            (kill-buffer response-buffer)))))))

** Parse

#+NAME: orga_core__parse-response
#+BEGIN_SRC emacs-lisp
(defun orga/parse-response (data)
  "Parse the decoded JSON ALIST `data` and return a plist of extracted fields.
This is a pure function that does not perform network operations or buffer manipulation."
  (let ((choices (cdr (assoc 'choices data))))
    (if (and choices (> (length choices) 0))
        (let* ((choice        (aref choices 0))
               (message-obj   (cdr (assoc 'message choice)))
               (content       (let ((s (cdr (assoc 'content message-obj))))
                                (when (and s
                                           (not (eq s :null))
                                           (not (string-empty-p s)))
                                  s)))
               (finish-reason (cdr (assoc 'finish_reason choice)))
               ;; -- tool_calls: present when finish_reason is "tool_calls" --
               (tool-calls    (let ((tc (cdr (assoc 'tool_calls message-obj))))
                                (and tc (not (eq tc :null))
                                     (> (length tc) 0)
                                     tc)))
               ;; -- reasoning: pull the first reasoning.text entry ------
               (reasoning
                (let* ((details (cdr (assoc 'reasoning_details message-obj)))
                       (text    (when (and details (> (length details) 0))
                                  (cdr (assoc 'text (aref details 0))))))
                  (when (and text (not (eq text :null))
                             (not (string-empty-p text)))
                    text)))
               ;; -- provider --------------------------------------------
               (provider
                (let ((p (cdr (assoc 'provider data))))
                  (when (and p (not (eq p :null))) p)))
               ;; -- usage -----------------------------------------------
               (usage              (cdr (assoc 'usage data)))
               (tokens-in          (cdr (assoc 'prompt_tokens usage)))
               (tokens-out         (cdr (assoc 'completion_tokens usage)))
               (tokens-reasoning   (cdr (assoc 'reasoning_tokens
                                                (cdr (assoc 'completion_tokens_details usage)))))
               (cost               (cdr (assoc 'cost usage))))
          (list :content          content
                :finish-reason    (when (and finish-reason
                                             (not (eq finish-reason :null)))
                                    finish-reason)
                :tool-calls       tool-calls
                :reasoning        reasoning
                :provider         provider
                :tokens-in        (when (and tokens-in (not (eq tokens-in :null)))
                                    tokens-in)
                :tokens-out       (when (and tokens-out (not (eq tokens-out :null)))
                                    tokens-out)
                :tokens-reasoning (when (and tokens-reasoning (not (eq tokens-reasoning :null)))
                                    tokens-reasoning)
                :cost             (when (and cost (not (eq cost :null)))
                                    cost)))
      (error "OrgA: empty choices in response"))))

** Call-Param Formatting

Dedicated block -- this is the seam where auto-eval of =#+CALL:= lines will eventually hook in,
so it stays decoupled from Commit's heading-writing logic rather than being folded into it.

#+NAME: orga_core__call-params
#+BEGIN_SRC emacs-lisp
(defun orga/call-param-literal (value)
  "Render VALUE as a literal suitable for a Babel #+CALL: argument.
Strings are double-quoted (with internal quotes/backslashes escaped, and physical
newlines safely serialized to literal \\n text); numbers, t/nil/:null, and other
atoms are rendered as bare literals. Vectors/lists are rendered as Elisp-readable strings."
  (cond
   ((stringp value)
    (format "\"%s\""
            (replace-regexp-in-string
             "\n" "\\\\n"  ;; Convert physical newline character into literal "\n" text
             (replace-regexp-in-string
              "\"" "\\\\\""
              (replace-regexp-in-string "\\\\" "\\\\\\\\" value)))))
   ((or (null value) (eq value :null)) "nil")
   ((eq value t) "t")
   ((eq value :false) "nil")
   ((numberp value) (format "%s" value))
   (t (format "\"%s\""
              (replace-regexp-in-string
               "\"" "\\\\\""
               (format "%S" value))))))

(defun orga/tool-args->call-params (fn-args)
  "Convert FN-ARGS (a JSON object string, e.g. \"{\\\"a\\\":1}\") into a
comma-separated Babel #+CALL: parameter list, e.g. a=1,b=\"text\".
Returns the empty string if FN-ARGS has no parseable key/value pairs."
  (let* ((json-object-type 'alist)
         (json-key-type    'symbol)
         (json-array-type  'vector)
         (parsed (ignore-errors (json-read-from-string (or fn-args "{}")))))
    (mapconcat
     (lambda (pair)
       (format "%s=%s" (car pair) (orga/call-param-literal (cdr pair))))
     parsed
     ",")))

** Commit

Writes the record of what the model said/asked for into the Org
buffer, *before* anything gets executed: =** Assistant= or
=** Assistant Tool Call= heading + metadata properties + optional
=:REASONING:= drawer, the
=** Tool= heading scaffolding with =#+CALL:= lines or placeholder
comments.

#+NAME: orga_core__commit
#+BEGIN_SRC emacs-lisp
(cl-defun orga/handle-response (response model timestamp dry-run)
  "Insert RESPONSE (from `orga/parse-response') into the buffer.
MODEL and TIMESTAMP are strings used in the inserted headings.
Dispatches on finish-reason: tool_calls → scaffold ** Tool headings;
anything else → insert ** Assistant reply and a fresh ** Input."
  (when (null response) (cl-return-from orga/handle-response nil))
  (let* ((ai-reply         (plist-get response :content))
         (finish-reason    (or (plist-get response :finish-reason) "unknown"))
         (tool-calls       (plist-get response :tool-calls))
         (reasoning        (plist-get response :reasoning))
         (provider         (or (plist-get response :provider) ""))
         (tokens-in        (plist-get response :tokens-in))
         (tokens-out       (plist-get response :tokens-out))
         (tokens-reasoning (plist-get response :tokens-reasoning))
         (cost             (plist-get response :cost)))

    (goto-char (point-max))
    (unless (bolp) (insert "\n"))

    (if (string= finish-reason "tool_calls")
        ;; -------------------------------------------------------------------
        ;; The model wants to call tools.
        ;; -------------------------------------------------------------------
        (progn
          ;; a. ** Assistant Tool Call heading + metadata
          ;;    :TOOL_CALLS_JSON: stores the raw array for lossless replay.
          (insert (format (concat "** Assistant Tool Call\n"
                                  ":PROPERTIES:\n"
                                  ":MODEL: %s\n"
                                  ":PROVIDER: %s\n"
                                  ":DATE: %s\n"
                                  ":FINISH_REASON: %s\n"
                                  ":TOKENS_IN: %s\n"
                                  ":TOKENS_OUT: %s\n"
                                  ":TOKENS_REASONING: %s\n"
                                  ":COST_USD: %s\n"
                                  ":TOOL_CALLS_JSON: %s\n"
                                  ":END:\n")
                          model provider timestamp finish-reason
                          (or tokens-in "")
                          (or tokens-out "")
                          (or tokens-reasoning "")
                          (if cost (format "%.6f" cost) "")
                          (json-encode tool-calls)))
          ;; b. Optional :REASONING: drawer
          (when (and reasoning (not (string-empty-p reasoning)))
            (insert (format ":REASONING:\n%s\n:END:\n" reasoning)))
          (save-excursion
            (re-search-backward "^\\*\\* Assistant Tool Call" nil t)
            (if (fboundp 'org-fold-hide-drawer-all)
                (org-fold-hide-drawer-all)
              (org-hide-drawer-all)))

          ;; c. One ** Tool heading per tool call — NO ** Input follows.
          (mapc
           (lambda (tc)
             (let* ((call-id    (or (cdr (assoc 'id tc)) ""))
                    (fn-alist   (cdr (assoc 'function tc)))
                    (fn-name    (or (cdr (assoc 'name fn-alist)) ""))
                    (fn-args    (or (cdr (assoc 'arguments fn-alist)) "{}"))
                    (impl-block (concat "orga_tool_implementation__" fn-name)))
               (goto-char (point-max))
               (unless (bolp) (insert "\n"))
               (insert (format (concat "** Tool\n"
                                       ":PROPERTIES:\n"
                                       ":TOOL_CALL_ID: %s\n"
                                       ":FUNCTION_NAME: %s\n"
                                       ":ARGUMENTS: %s\n"
                                       ":END:\n")
                               call-id fn-name fn-args))
               (save-excursion
                 (re-search-backward "^\\*\\* Tool" nil t)
                 (if (fboundp 'org-fold-hide-drawer-all)
                     (org-fold-hide-drawer-all)
                   (org-hide-drawer-all)))
               (if (save-excursion
                     (goto-char (point-min))
                     (re-search-forward
                      (concat "#\\+NAME:[ \t]+" (regexp-quote impl-block)) nil t))
                   (insert (format "#+CALL: %s(%s)\n"
                                    impl-block
                                    (orga/tool-args->call-params fn-args)))
                 (insert (format "# No implementation for %s -- paste result below:\n"
                                 fn-name)))))
           (append tool-calls nil))          ; vector → list

          (org-reveal)
          (goto-char (point-max))
          (message "OrgA: tool call(s) received -- fill in ** Tool sections, then C-c C-v to continue."))

      ;; ---------------------------------------------------------------------
      ;; Regular assistant text reply — append heading + fresh ** Input.
      ;; ---------------------------------------------------------------------
      (insert (format (concat "** Assistant\n"
                              ":PROPERTIES:\n"
                              ":MODEL: %s\n"
                              ":PROVIDER: %s\n"
                              ":DATE: %s\n"
                              ":FINISH_REASON: %s\n"
                              ":TOKENS_IN: %s\n"
                              ":TOKENS_OUT: %s\n"
                              ":TOKENS_REASONING: %s\n"
                              ":COST_USD: %s\n"
                              ":END:\n")
                      model provider timestamp finish-reason
                      (or tokens-in "")
                      (or tokens-out "")
                      (or tokens-reasoning "")
                      (if cost (format "%.6f" cost) "")))
      (when (and reasoning (not (string-empty-p reasoning)))
        (insert (format ":REASONING:\n%s\n:END:\n" reasoning)))
      (insert (format "%s\n" ai-reply))
      (save-excursion
        (re-search-backward "^\\*\\* Assistant" nil t)
        (if (fboundp 'org-fold-hide-drawer-all)
            (org-fold-hide-drawer-all)
          (org-hide-drawer-all)))
      (goto-char (point-max))
      (unless (bolp) (insert "\n"))
      (unless orga/dry-run (insert "** Input\n"))
      (org-reveal)
      (goto-char (point-max))
      (message "OrgA: reply received."))))

  "Fraction of the model context window above which auto-compaction fires.")

;; ---- probes ---------------------------------------------------------------
(defun orga/model-context-window (model)
  "Return MODEL's context window in tokens, querying OpenRouter once.
Prefers the serving provider's ceiling (top_provider.context_length),
falling back to the model-level context_length.  Memoized; nil on failure."
  (or (cdr (assoc model orga/context-window-cache))
      (let ((curl (executable-find "curl")))
        (when curl
          (with-temp-buffer
            (when (zerop (call-process curl nil t nil
                                       "-s" "--max-time" "20"
                                       "https://openrouter.ai/api/v1/models"))
              (goto-char (point-min))
              (let* ((json-object-type 'alist)
                     (json-array-type  'vector)
                     (data   (ignore-errors (json-read)))
                     (models (cdr (assoc 'data data)))
                     (win nil))
                (when models
                  (cl-loop for m across models
                           when (equal (cdr (assoc 'id m)) model) do
                           (let* ((tp  (cdr (assoc 'top_provider m)))
                                  (tpc (and tp (cdr (assoc 'context_length tp))))
                                  (mc  (cdr (assoc 'context_length m))))
                             (setq win (or (and tpc (not (eq tpc :null)) tpc)
                                           (and mc  (not (eq mc :null))  mc))))
                           and return nil))
                (when win (push (cons model win) orga/context-window-cache))
                win)))))))

(defun orga/current-context-tokens ()
  "Return the most recent :TOKENS_IN: recorded in the buffer, or nil.
This is the size of the last prompt we sent -- the best available proxy
for current context pressure."
  (save-excursion
    (goto-char (point-max))
    (when (re-search-backward "^[ \t]*:TOKENS_IN:[ \t]*\\([0-9]+\\)" nil t)
      (string-to-number (match-string 1)))))

;; ---- turn grouping --------------------------------------------------------
(defun orga/chat-turns (from)
  "Return turn groups in * Chat History from buffer position FROM onward.
Each group is a plist (:beg :end :last :kind).  An Assistant Tool Call and
all its Tool replies collapse into ONE group, so a cutoff never splits a
call from its results.  The trailing ** Input draft is excluded."
  (save-excursion
    (let ((section-end (save-excursion
                         (goto-char (point-min))
                         (re-search-forward "^\\* Chat History" nil t)
                         (org-end-of-subtree t) (point)))
          headings)
      (goto-char from)
      (while (re-search-forward
              "^\\*\\* \\(User\\|Input\\|Assistant Tool Call\\|Assistant\\|Tool\\)"
              section-end t)
        (push (list :pos  (match-beginning 0)
                    :kind (match-string 1)
                    :end  (save-excursion (goto-char (match-beginning 0))
                                          (org-end-of-subtree t) (point)))
              headings))
      (setq headings (nreverse headings))
      (let (groups cur)
        (dolist (h headings)
          (let ((kind (plist-get h :kind))
                (pos  (plist-get h :pos))
                (end  (plist-get h :end)))
            (cond
             ((string= kind "Tool")
              (if (and cur (eq (plist-get cur :kind) 'tool-group))
                  (setq cur (plist-put (plist-put cur :end end) :last pos))
                (when cur (push cur groups))
                (setq cur (list :beg pos :end end :last pos :kind 'tool-group))))
             ((string= kind "Assistant Tool Call")
              (when cur (push cur groups))
              (setq cur (list :beg pos :end end :last pos :kind 'tool-group)))
             ((string= kind "Input")
              (when cur (push cur groups) (setq cur nil)))
             (t ; User or Assistant text
              (when cur (push cur groups) (setq cur nil))
              (push (list :beg pos :end end :last pos :kind 'plain) groups)))))
        (when cur (push cur groups))
        (nreverse groups)))))

(defun orga/compact-context (&optional verbose)
  "Fold the oldest chat turns into the cumulative * Summary block.
Keeps `orga/keep-recent-turns' turns live, summarizes the rest together
with the existing summary via `orga/summary-model' (or the main model),
overwrites * Summary, and advances the :cutoff: tag.  Always makes a live
API call, so it is refused while dry-run is ON.  Returns t on success."
  (interactive "p")
  (when (and (called-interactively-p 'any) orga/dry-run)
    (user-error "OrgA: compaction makes a live API call; disabled while dry-run is ON"))
  (let* ((old-cutoff
          (save-excursion
            (goto-char (point-min))
            (re-search-forward "^\\* Chat History" nil t)
            (let ((section-end (save-excursion (org-end-of-subtree t) (point)))
                  pos)
              (while (re-search-forward "^\\*\\* .*:cutoff:" section-end t)
                (setq pos (match-beginning 0)))
              pos)))
         (from (if old-cutoff
                   (save-excursion (goto-char old-cutoff)
                                   (org-end-of-subtree t) (point))
                 (save-excursion (goto-char (point-min))
                                 (re-search-forward "^\\* Chat History" nil t)
                                 (point))))
         (groups (orga/chat-turns from))
         (keep   (max 1 orga/keep-recent-turns))
         (fold-count (- (length groups) keep)))
    (if (<= fold-count 0)
        (prog1 nil
          (when verbose
            (message "OrgA: nothing to compact (%d turn(s), keeping %d)."
                     (length groups) keep)))
      (let* ((fold-groups   (cl-subseq groups 0 fold-count))
             (fold-beg      (plist-get (car fold-groups) :beg))
             (fold-end      (plist-get (car (last fold-groups)) :end))
             (new-cutoff    (plist-get (car (last fold-groups)) :last))
             (folded-text   (buffer-substring-no-properties fold-beg fold-end))
             (old-summary
              (save-excursion
                (goto-char (point-min))
                (when (re-search-forward "^\\* Summary" nil t)
                  (let ((b (progn (forward-line 1) (point)))
                        (e (progn (org-end-of-subtree t) (point))))
                    (string-trim (buffer-substring-no-properties b e))))))
             (messages
              (vector
               `((role . "developer")
                 (content . ,(concat
                              "You are a context-compaction engine. Produce a "
                              "dense, faithful summary preserving facts, "
                              "decisions, names, identifiers, file paths, open "
                              "questions, and unfinished tasks. Merge the "
                              "existing summary with the new turns into one "
                              "cumulative summary. Omit pleasantries. Plain prose.")))
               `((role . "user")
                 (content . ,(concat
                              (when (and old-summary (not (string-empty-p old-summary)))
                                (concat "Existing summary:\n" old-summary "\n\n"))
                              "New conversation turns to fold in:\n\n"
                              folded-text)))))
             (model  (or orga/summary-model <<openrouter_model>>))
             (raw    (orga/send-request model messages nil nil))
             (parsed (and raw (orga/parse-response raw)))
             (summary (and parsed (plist-get parsed :content))))
        (if (or (null summary) (string-empty-p summary))
            (prog1 nil
              (message "OrgA: compaction aborted -- summarizer returned no text."))
          (orga/write-summary summary)
          (orga/move-cutoff-to new-cutoff)
          (message "OrgA: compacted %d turn(s); %d kept live." fold-count keep)
          t)))))

** Cycle Driver

The orchestrator: decides which collect mode applies (pending tool
results vs fresh =** Input=), then walks collect -> assemble -> send ->
parse -> commit -> execute -> yield in order. Doesn't belong to any
single stage, so it stays its own block, defined last since it's the
only thing that calls all the others.

#+NAME: orga_core__cycle-driver
#+BEGIN_SRC emacs-lisp :noweb yes
(defun orga/pending-tool-results-p ()
  "Return non-nil when the chat has tool results ready to send.
Specifically: the last '** Assistant Tool Call' heading in * Chat History
is followed by at least one '** Tool' sibling, and every such sibling has
a non-empty body (i.e. the user has filled in all results).

Returns nil when the history is empty, has no tool calls, or when any
Tool result body is still empty."
  (cl-block orga/pending-tool-results-p
    (save-excursion
      (goto-char (point-min))
      (unless (re-search-forward "^\\* Chat History" nil t)
        (cl-return-from orga/pending-tool-results-p nil))
      (let* ((section-end (save-excursion (org-end-of-subtree t) (point)))
             ;; Find the last ** Assistant Tool Call in the section
             (last-tc-pos
              (save-excursion
                (let (pos)
                  (while (re-search-forward "^\\*\\* Assistant Tool Call" section-end t)
                    (setq pos (match-beginning 0)))
                  pos))))
        ;; No tool call heading at all → not in tool-result mode
        (unless last-tc-pos
          (cl-return-from orga/pending-tool-results-p nil))
        ;; If a ** Input heading exists after the last tool call, we're back
        ;; in normal input mode → not pending tool results
        (when (save-excursion
                (goto-char last-tc-pos)
                (org-end-of-subtree t)
                (re-search-forward "^\\*\\* Input" section-end t))
          (cl-return-from orga/pending-tool-results-p nil))
        ;; Collect ** Tool headings that are siblings after the last tool call
        (let* ((tc-end (save-excursion
                         (goto-char last-tc-pos)
                         (org-end-of-subtree t)
                         (point)))
               (tool-headings
                (save-excursion
                  (goto-char tc-end)
                  (let (positions)
                    (while (re-search-forward "^\\*\\* Tool" section-end t)
                      (push (match-beginning 0) positions))
                    (nreverse positions)))))
          ;; There must be at least one Tool heading, and every one must be filled
          (and tool-headings
               (cl-every
                (lambda (pos)
                  (save-excursion
                    (goto-char pos)
                    (not (string-empty-p (orga/subtree-body)))))
                tool-headings)))))))

(defun ai-chat-cycle ()
  "Send the current turn to the AI and append the reply.

Two modes are detected automatically:

NORMAL MODE — a ** Input heading exists with content:
  The heading is committed as a dated ** User message, the API is called,
  and a ** Assistant reply is appended followed by a fresh ** Input.

TOOL-RESULT MODE — the last ** Assistant Tool Call heading is present and
  every subsequent ** Tool sibling has a non-empty body:
  No ** Input conversion takes place.  The filled tool results are already
  part of the history (via orga/collect-history).  The API is called and
  the reply is appended followed by a fresh ** Input.

When the model itself responds with tool calls (finish_reason = tool_calls):
  - A ** Assistant Tool Call heading is inserted with :TOOL_CALLS_JSON:.
  - One ** Tool heading per call with :TOOL_CALL_ID:, :FUNCTION_NAME:,
    :ARGUMENTS: and either a #+CALL: line or a placeholder comment.
  - No ** Input is appended; C-c C-v again after filling the results.

Bound to C-c C-v by the file-local eval cookie at the top of this file."
  (interactive)
  (let ((model     <<openrouter_model>>)
        (timestamp (format-time-string "%Y-%m-%d %H:%M")))
    (orga/maybe-auto-compact model)
    (cond

     ;; -----------------------------------------------------------------------
     ;; TOOL-RESULT MODE: all ** Tool bodies are filled — send results to API.
     ;; -----------------------------------------------------------------------
     ((orga/pending-tool-results-p)
      (message (if orga/dry-run "OrgA: dry run..." "OrgA: sending tool results..."))
      (let* ((context  (orga/create-context))
             (tools    (orga/collect-tools))
             (raw      (orga/send-request model context tools orga/dry-run))
             (response (and raw (orga/parse-response raw))))
        (orga/handle-response response model timestamp orga/dry-run)))

     ;; -----------------------------------------------------------------------
     ;; NORMAL MODE: commit ** Input as ** User and call API.
     ;; -----------------------------------------------------------------------
     (t
      (pcase (orga/read-input-heading)
        ('nil
         (message "OrgA: nothing to send (no ** Input and no pending tool results)."))
        (`(,start ,end ,user-text)
         (when (string-empty-p user-text)
           (user-error "OrgA: ** Input is empty -- write something first"))
         ;; 1. Overwrite the ** Input subtree with a dated ** User entry
         (delete-region start end)
         (goto-char (point-max))
         (unless (bolp) (insert "\n"))
         (insert (format "** User\n:PROPERTIES:\n:DATE: %s\n:END:\n%s\n"
                         timestamp user-text))
         ;; 2. Call API
         (message (if orga/dry-run "OrgA: dry run..." "OrgA: sending..."))
         (let* ((context  (orga/create-context))
                (tools    (orga/collect-tools))
                (raw      (orga/send-request model context tools orga/dry-run))
                (response (and raw (orga/parse-response raw))))
           (orga/handle-response response model timestamp orga/dry-run))))))))

#+NAME: orga_tool_implementation__execute_shell_command
#+BEGIN_SRC emacs-lisp :var command="pwd" :var cwd="./" :var timeout_sec=30 :results output
(let* ((process-name "orga-persistent-shell")
       (buffer-name (concat "*" process-name "*"))
       (prompt-marker "__ORGA_PROMPT__")
       (proc-raw (get-process process-name))
       (proc-alive (and proc-raw (process-live-p proc-raw)))
       (proc (or (and proc-alive proc-raw)
                 (let ((default-directory (expand-file-name cwd)))
                   (make-comint-in-buffer process-name buffer-name
                                           "/bin/bash" nil "--norc" "--noprofile")
                   (get-process process-name)))))

  (cl-flet ((wait-for-prompt (timeout)
              (let ((start (float-time)))
                (while (and (process-live-p proc)
                            (< (- (float-time) start) timeout)
                            (not (save-excursion
                                   (goto-char (point-max))
                                   (forward-line -1)
                                   (looking-at-p
                                    (concat "^" (regexp-quote prompt-marker) "$")))))
                  (accept-process-output proc 0.1)))))

    (with-current-buffer (process-buffer proc)
      ;; 1. Initialize prompt and turn off echoes for a fresh shell
      (unless proc-alive
        (process-send-string
         proc (format "stty -echo\nexport PS1='%s\\n'\n" prompt-marker))
        (wait-for-prompt 2))

      ;; 2. Synchronize directory
      (process-send-string
       proc (format "cd %s\n" (shell-quote-argument cwd)))
      (wait-for-prompt 5)

      ;; 3. Run command and wait for the exit sentinel in the NEW region
      (goto-char (point-max))
      (let ((start-marker (point-marker)))
        (process-send-string
         proc (concat command "\n" "echo \"__ORGA_EXIT__:$?\"\n"))
        (let ((start (float-time)))
          (while (and (process-live-p proc)
                      (< (- (float-time) start) timeout_sec)
                      (not (save-excursion
                             (goto-char start-marker)

 ;; Find the named src block and return its body.
 ((string-prefix-p "wiki://" uri)
  (let ((block-name uri)
        (found nil))
    (save-excursion
      (goto-char (point-min))
      (while (and (not found)
                  (re-search-forward
                   (concat "#\\+NAME:[ \t]+" (regexp-quote block-name))
                   nil t))
        ;; Advance past the #+NAME: line to the #+BEGIN_SRC line
        (when (re-search-forward "#\\+BEGIN_SRC" nil t)
          (forward-line 1)
          (let* ((beg (point))
                 (end (progn
                        (re-search-forward "#\\+END_SRC" nil t)
                        (match-beginning 0)))
                 (body (string-trim (buffer-substring-no-properties beg end))))
            (setq found body)))))
    (if found
        (princ found)
      (princ (format "ERROR: no wiki block found for URI: %s" uri)))))

 ;; ── chat://heading-title/n ──────────────────────────────────────────────────
 ;; Find a top-level heading matching heading-title, then return the Nth
 ;; child subtree body (1-based).
 ((string-prefix-p "chat://" uri)
  (let* ((path    (substring uri (length "chat://")))
         (slash   (string-match "/" path))
         (heading (decode-coding-string (url-unhex-string (if slash (substring path 0 slash) path)) 'utf-8))
         (n-str   (if slash (substring path (1+ slash)) "1"))
         (n       (string-to-number n-str))
         (found   nil))
    (save-excursion
      (goto-char (point-min))
      (when (re-search-forward
             (concat "^\\* " (regexp-quote heading)) nil t)
        (let ((section-end (save-excursion (org-end-of-subtree t) (point)))
              (count 0))
          (while (and (not found)
                      (re-search-forward "^\\*\\* " section-end t))
            (setq count (1+ count))
            (when (= count n)
              (setq found (orga/subtree-body)))))))
    (if found
        (princ found)
      (princ (format "ERROR: could not find child %s of heading '%s'" n-str heading)))))

 ;; ── https:// URL ────────────────────────────────────────────────────────────
 ;; Fetch with curl, strip HTML tags, collapse whitespace, return plain text.
 ((string-prefix-p "https://" uri)
  (let ((curl (executable-find "curl")))
    (unless curl
      (error "fetch: curl not found"))
    (let* ((raw (with-output-to-string
                  (with-current-buffer standard-output
                    (call-process curl nil t nil
                                  "-s" "--max-time" "30"
                                  "-L"
                                  "-A" "Mozilla/5.0 (X11; Linux x86_64)"
                                  uri))))
           ;; Remove <script>…</script> and <style>…</style> blocks first
           (no-scripts (replace-regexp-in-string
                        "<script[^>]*>\\(?:.\\|\n\\)*?</script>" "" raw))
           (no-styles  (replace-regexp-in-string
                        "<style[^>]*>\\(?:.\\|\n\\)*?</style>" "" no-scripts))
           ;; Strip remaining tags
           (no-tags    (replace-regexp-in-string "<[^>]+>" "" no-styles))
           ;; Decode common HTML entities
           (decoded    (let ((s no-tags))
                         (dolist (pair '(("&amp;"  . "&")  ("&lt;"   . "<")
                                        ("&gt;"   . ">")  ("&quot;" . "\"")
                                        ("&#x27;" . "'")  ("&#39;"  . "'")
                                        ("&nbsp;" . " ")))
                           (setq s (replace-regexp-in-string
                                    (car pair) (cdr pair) s t t)))
                         s))
           ;; Collapse runs of blank lines to one blank line
           (clean      (replace-regexp-in-string
                        "\n\\{3,\\}" "\n\n"
                        (string-trim decoded))))
      (if (string-empty-p clean)
          (princ (format "ERROR: no content retrieved from %s" uri))
        ;; Cap output to ~8000 chars to avoid flooding the context window
        (princ (if (> (length clean) 8000)
                   (concat (substring clean 0 8000) "\n…[truncated]")
                 clean))))))

 (t
  (princ (format "ERROR: unsupported URI scheme: %s" uri))))

## Before We Begin

## Summary

OrgA is an [[wiki://concepts/org-mode]|Org-mode]]-native [[wiki://concepts/llm|LLM]] agent. It runs entirely
inside a single `.org` file: the [[wiki://concepts/emacs-lisp|Emacs Lisp]] runtime, tool definitions,
chat history, and system prompt all cohabit the same buffer. Conversation is driven by the
`orga/send` function, which collects context, calls an
[[wiki://concepts/openrouter|OpenRouter]]-compatible API, and writes the reply back into the file under
a `** Assistant` heading in Chat History.

## Runtime

The agent is initialized by evaluating a named [[wiki://concepts/org-babel|Org Babel]] source block on
file open (see the `eval:` directive in the file's first line). All logic lives in `emacs-lisp`
src blocks grouped under `* Initialize AI Sender Method`.

Context sent to the API on each turn is assembled from three sources:

- **System Prompt** — the developer-role message, extracted from the
  `#+BEGIN_SRC markdown` block inside `* System Prompt`.
- **Summary** — a condensed recap of turns above the `:cutoff:` marker, sent
  as a user message to preserve older context cheaply.
- **Chat History** — live `** User`, `** Assistant`,
  `** Assistant Tool Call`, and `** Tool` headings, collected in order.

## Tools

Tools are declared under `* Tools`. Each tool subtree holds three artefacts: a human-readable
JSON schema, an [[wiki://concepts/emacs-lisp|Emacs Lisp]] alist schema
(named `orga_tool_schema__<name>`), and an implementation src block
(named `orga_tool_implementation__<name>`). A `:disabled:` tag on the heading excludes the tool
from the API call without deleting it.

## Behaviour

- Responds in natural prose; [[wiki://concepts/markdown|Markdown]] formatting only when explicitly
  requested.
- Never begins a line with `*`, which would be misread as an Org-mode heading.
- Replies are inserted under `** Assistant` in Chat History.
- The Orga agent MAY follow a wikilink, such as [[wiki://path/to/some/concept|Some Concept]],
  if it is interested in exploring a topic mentioned in the context.
- However, if one or more unvisited wikilink(s) exist that are directly necessary to understand the
  current context, the Orga agent SHOULD follow those wikilink(s) before replying.

OpenRouter is a unified API platform that gives you access to over 400 AI models from dozens of providers through a single endpoint. Instead of juggling separate API keys for OpenAI, Anthropic, Google, Meta, and others, you use one key to reach their entire model catalog.
The platform works as an intelligent router, sending your requests to the right provider while taking care of authentication, billing, and error handling. This approach fixes several headaches that come with using multiple AI providers.
Problems OpenRouter solves
Working with multiple AI providers gets messy fast. Each one has its own API format, login process, and billing system. You end up maintaining separate code for each service, which slows down development and makes testing new models a pain.
Things get worse when providers go down or hit you with rate limits. Your app breaks, and there’s nothing you can do except wait. Plus, figuring out which provider offers the best price for similar models means tracking costs manually across different platforms.
The biggest issue is getting locked into one provider. When you build everything around their specific API, switching to better models or cheaper options later becomes a major project.
How OpenRouter fixes this
OpenRouter solves these problems with a set of connected features:

One API key works with 400+ models from all major providers
Automatic switching to backup providers when your first choice fails
Side-by-side pricing for all models so you can compare costs instantly
Works with existing OpenAI code — just change the endpoint URL
Real-time monitoring that routes requests to the fastest available provider

These pieces work together to make AI development smoother and more reliable.
Who should use OpenRouter?
Different types of users get value from this unified approach:

Developers can try new models without setting up accounts everywhere, making experimentation faster
Enterprise teams get the uptime they need through automatic backups when providers fail
Budget-conscious users can find the cheapest option for their needs without spreadsheet math
Researchers get instant access to cutting-edge models without account setup overhead

Prerequisites
Before diving into OpenRouter, you’ll need a few things set up on your machine. This tutorial assumes you’re comfortable with basic Python programming and have worked with APIs before. You don’t need to be an expert, but you should understand concepts like making HTTP requests and handling JSON responses.
You’ll need Python 3.7 or later installed on your system. We’ll be using the openai Python package to interact with OpenRouter's API, along with python-dotenv to handle environment variables securely. You can install both with:
pip install requests openai python-dotenv
You’ll also need an OpenRouter account and API key. Head to openrouter.ai to create a free account — you’ll get a small credit allowance to test things out. Once you’re logged in, go to the API Keys section in your account settings and generate a new key.
After getting your API key, create a .env file in your project directory and add your key like this:
OPENROUTER_API_KEY=your_api_key_here
This keeps your API key secure and out of your code. If you plan to use OpenRouter beyond testing, you’ll need to add credits to your account through the Credits page.
With these basics in place, you’re ready to make your first API call through OpenRouter.
Making Your First API Call in OpenRouter
Getting started with OpenRouter is remarkably simple if you’ve used the OpenAI SDK before. You just change one line of code and suddenly have access to hundreds of models from different providers.
Your first request and setup
Let’s jump right in with a working example that demonstrates OpenRouter’s approach:
import os
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()

client = OpenAI(
   base_url="https://openrouter.ai/api/v1",
   api_key=os.getenv("OPENROUTER_API_KEY"),
)

response = client.chat.completions.create(
   model="openai/gpt-5-mini",
   messages=[
       {
           "role": "user",
           "content": "Write a haiku about debugging code at 2 AM"
       }
   ]
)

print(response.choices[0].message.content)
Night hum, coffee cooled
cursor blinks, bug hides somewhere
I chase ghosts 'til dawn
The magic happens in two places. First, the base_url parameter redirects your requests to OpenRouter's servers instead of OpenAI's. Second, the model name follows a provider/model-name format - openai/gpt-5-mini instead of just gpt-5-mini. This tells OpenRouter which provider's version you want while keeping the familiar interface. Here are some common models that you can plug into the above example without any errors:

google/gemini-2.5-pro
mistralai/mistral-nemo
